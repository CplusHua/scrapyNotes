function getprice(code,id,oldprice)
{
//code = document.fjzt_form.stock_num.value;
//alert(code);
if(code.length!=6 || code.match(/^[0-9,]*$/)==null) return;

var xmlHttp;
xmlHttp = xmlhttpinit();
xmlHttp.onreadystatechange=function()
{
if(xmlHttp.readyState==4){
var ss = xmlHttp.responseText.split(",");
var price;
var zf;
if (ss[0] == 0){
price = "停盘";
} else {
price = ss[0];
zf = (100*(ss[0]-oldprice)/oldprice).toFixed(2);
document.getElementById('zf' + id).innerHTML = zf + "%";
}
document.getElementById('s' + id).innerHTML = price;
document.getElementById('s' + id).parentNode.className = "tbn";
//			document.getElementById('zf' + id).innerHTML = zf + "%";
document.getElementById('zf' + id).parentNode.className = "tbn";
document.getElementById('zf' + id).parentNode.style.textAlign = "center";
if(zf < 0){
document.getElementById('zf' + id).parentNode.parentNode.style.color="#090";
}
if(zf > 0){
document.getElementById('zf' + id).parentNode.parentNode.style.color="#f00";
}

//			document.write(ss[0]);
//			alert(this.innerHTML);
//			document.moderate.c_price=ss[0];
//			document.fjzt_form.stock_name.value=ss[1];
//			alert(ss[0]+ss[1]);
}
}
xmlHttp.open("GET","\getstock.php?n="+code,true);
//xmlHttp.open("GET","http://wwww.178448.com/getstock.php?n="+code,true);
xmlHttp.send(null);
}
